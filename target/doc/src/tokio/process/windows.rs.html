1//! Windows asynchronous process handling.
2//!
3//! Like with Unix we don't actually have a way of registering a process with an
4//! IOCP object. As a result we similarly need another mechanism for getting a
5//! signal when a process has exited. For now this is implemented with the
6//! `RegisterWaitForSingleObject` function in the kernel32.dll.
7//!
8//! This strategy is the same that libuv takes and essentially just queues up a
9//! wait for the process in a kernel32-specific thread pool. Once the object is
10//! notified (e.g. the process exits) then we have a callback that basically
11//! just completes a `Oneshot`.
12//!
13//! The `poll_exit` implementation will attempt to wait for the process in a
14//! nonblocking fashion, but failing that it'll fire off a
15//! `RegisterWaitForSingleObject` and then wait on the other end of the oneshot
16//! from then on out.
17

59struct Waiting {

68pub(crate) fn build_child(mut child: StdChild) -> io::Result<SpawnedChild> {
69    let stdin = child.stdin.take().map(stdio).transpose()?;
70    let stdout = child.stdout.take().map(stdio).transpose()?;
71    let stderr = child.stderr.take().map(stdio).transpose()?;
72
73    Ok(SpawnedChild {
74        child: Child {
75            child,
76            waiting: None,
77        },
78        stdin,
79        stdout,
80        stderr,
81    })
82}

100impl Future for Child {
101    type Output = io::Result<ExitStatus>;
102
103    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
104        let inner = Pin::get_mut(self);
105        loop {
106            if let Some(ref mut w) = inner.waiting {
107                match Pin::new(&mut w.rx).poll(cx) {
108                    Poll::Ready(Ok(())) => {}
109                    Poll::Ready(Err(_)) => panic!("should not be canceled"),
110                    Poll::Pending => return Poll::Pending,
111                }
112                let status = inner.try_wait()?.expect("not ready yet");
113                return Poll::Ready(Ok(status));
114            }
115
116            if let Some(e) = inner.try_wait()? {
117                return Poll::Ready(Ok(e));
118            }
119            let (tx, rx) = oneshot::channel();
120            let ptr = Box::into_raw(Box::new(Some(tx)));
121            let mut wait_object = null_mut();
122            let rc = unsafe {
123                RegisterWaitForSingleObject(
124                    &mut wait_object,
125                    inner.child.as_raw_handle() as _,
126                    Some(callback),
127                    ptr as *mut _,
128                    INFINITE,
129                    WT_EXECUTEINWAITTHREAD | WT_EXECUTEONLYONCE,
130                )
131            };
132            if rc == 0 {
133                let err = io::Error::last_os_error();
134                drop(unsafe { Box::from_raw(ptr) });
135                return Poll::Ready(Err(err));
136            }
137            inner.waiting = Some(Waiting {
138                rx,
139                wait_object,
140                tx: ptr,
141            });
142        }
143    }
144}

152impl Drop for Waiting {
153    fn drop(&mut self) {
154        unsafe {
155            let rc = UnregisterWaitEx(self.wait_object, INVALID_HANDLE_VALUE);
156            if rc == 0 {
157                panic!("failed to unregister: {}", io::Error::last_os_error());
158            }
159            drop(Box::from_raw(self.tx));
160        }
161    }
162}

164unsafe extern "system" fn callback(ptr: *mut std::ffi::c_void, _timer_fired: bool) {
165    let complete = unsafe { &mut *(ptr as *mut Option<oneshot::Sender<()>>) };
166    let _ = complete.take().unwrap().send(());
167}

236pub(super) fn stdio<T>(io: T) -> io::Result<ChildStdio>
237where
238    T: IntoRawHandle,
239{
240    use std::os::windows::prelude::FromRawHandle;
241
242    let raw = Arc::new(unsafe { StdFile::from_raw_handle(io.into_raw_handle()) });
243    let io = ArcFile(raw.clone());
244    // SAFETY: the `Read` implementation of `io` does not
245    // read from the buffer it is borrowing and correctly
246    // reports the length of the data written into the buffer.
247    let io = unsafe { Blocking::new(io) };
248    Ok(ChildStdio { raw, io })
249}

251fn convert_to_file(child_stdio: ChildStdio) -> io::Result<StdFile> {
252    let ChildStdio { raw, io } = child_stdio;
253    drop(io); // Try to drop the Arc count here
254
255    Arc::try_unwrap(raw).or_else(|raw| duplicate_handle(&*raw))
256}
257
258pub(crate) fn convert_to_stdio(child_stdio: ChildStdio) -> io::Result<Stdio> {
259    convert_to_file(child_stdio).map(Stdio::from)
260}

262fn duplicate_handle<T: AsRawHandle>(io: &T) -> io::Result<StdFile> {
263    use std::os::windows::prelude::FromRawHandle;
264
265    unsafe {
266        let mut dup_handle = INVALID_HANDLE_VALUE;
267        let cur_proc = GetCurrentProcess();
268
269        let status = DuplicateHandle(
270            cur_proc,
271            io.as_raw_handle() as _,
272            cur_proc,
273            &mut dup_handle,
274            0,
275            0,
276            DUPLICATE_SAME_ACCESS,
277        );
278
279        if status == 0 {
280            return Err(io::Error::last_os_error());
281        }
282
283        Ok(StdFile::from_raw_handle(dup_handle as _))
284    }
