17impl Drop for SecurityContext {
18    fn drop(&mut self) {
19        unsafe {
20            Identity::DeleteSecurityContext(&self.0);
21        }
22    }

39impl SecurityContext {
40    pub fn initialize(
41        cred: &mut SchannelCred,
42        accept: bool,
43        domain: Option<&[u16]>,
44        requested_application_protocols: &Option<Vec<Vec<u8>>>,
45    ) -> io::Result<(SecurityContext, Option<ContextBuffer>)> {
46        unsafe {
47            let mut ctxt = mem::zeroed();
48
49            if accept {
50                // If we're performing an accept then we need to wait to call
51                // `AcceptSecurityContext` until we've actually read some data.
52                return Ok((SecurityContext(ctxt), None));
53            }
54
55            let domain = domain.map(|b| b.as_ptr()).unwrap_or(ptr::null_mut());
56
57            let mut inbufs = vec![];
58
59            // Make sure `AlpnList` is kept alive for the duration of this function.
60            let mut alpns = requested_application_protocols
61                .as_ref()
62                .map(|alpn| AlpnList::new(alpn));
63            if let Some(ref mut alpns) = alpns {
64                inbufs.push(secbuf(
65                    Identity::SECBUFFER_APPLICATION_PROTOCOLS,
66                    Some(&mut alpns[..]),
67                ));
68            };
69
70            let inbuf_desc = secbuf_desc(&mut inbufs[..]);
71
72            let mut outbuf = [secbuf(Identity::SECBUFFER_EMPTY, None)];
73            let mut outbuf_desc = secbuf_desc(&mut outbuf);
74
75            let mut attributes = 0;
76
77            match Identity::InitializeSecurityContextW(
78                &cred.as_inner(),
79                ptr::null_mut(),
80                domain,
81                INIT_REQUESTS,
82                0,
83                0,
84                &inbuf_desc,
85                0,
86                &mut ctxt,
87                &mut outbuf_desc,
88                &mut attributes,
89                ptr::null_mut(),
90            ) {
91                Foundation::SEC_I_CONTINUE_NEEDED => {
92                    Ok((SecurityContext(ctxt), Some(ContextBuffer(outbuf[0]))))
93                }
94                err => Err(io::Error::from_raw_os_error(err)),
95            }
96        }
97    }
98
99    unsafe fn attribute<T>(&self, attr: Identity::SECPKG_ATTR) -> io::Result<T> {
100        let mut value = mem::zeroed();
101        let status =
102            Identity::QueryContextAttributesW(&self.0, attr, &mut value as *mut _ as *mut _);
103        match status {
104            Foundation::SEC_E_OK => Ok(value),
105            err => Err(io::Error::from_raw_os_error(err)),
106        }
107    }

109    pub fn application_protocol(&self) -> io::Result<Identity::SecPkgContext_ApplicationProtocol> {

113    pub fn session_info(&self) -> io::Result<Identity::SecPkgContext_SessionInfo> {

115    }

121    pub fn remote_cert(&self) -> io::Result<CertContext> {
122        unsafe {
123            self.attribute(Identity::SECPKG_ATTR_REMOTE_CERT_CONTEXT)
124                .map(|p| CertContext::from_inner(p))
125        }
126    }
127
128    pub fn local_cert(&self) -> io::Result<CertContext> {
129        unsafe {
130            self.attribute(Identity::SECPKG_ATTR_LOCAL_CERT_CONTEXT)
131                .map(|p| CertContext::from_inner(p))
132        }
133    }
