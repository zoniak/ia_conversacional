30/// Non-blocking windows named pipe.
31///
32/// This structure internally contains a `HANDLE` which represents the named
33/// pipe, and also maintains state associated with the mio event loop and active
34/// I/O operations that have been scheduled to translate IOCP to a readiness
35/// model.
36///
37/// Note, IOCP is a *completion* based model whereas mio is a *readiness* based
38/// model. To bridge this, `NamedPipe` performs internal buffering. Writes are
39/// written to an internal buffer and the buffer is submitted to IOCP. IOCP
40/// reads are submitted using internal buffers and `NamedPipe::read` reads from
41/// this internal buffer.
42///
43/// # Trait implementations
44///
45/// The `Read` and `Write` traits are implemented for `NamedPipe` and for
46/// `&NamedPipe`. This represents that a named pipe can be concurrently read and
47/// written to and also can be read and written to at all. Typically a named
48/// pipe needs to be connected to a client before it can be read or written,
49/// however.
50///
51/// Note that for I/O operations on a named pipe to succeed then the named pipe
52/// needs to be associated with an event loop. Until this happens all I/O
53/// operations will return a "would block" error.
54///
55/// # Managing connections
56///
57/// The `NamedPipe` type supports a `connect` method to connect to a client and
58/// a `disconnect` method to disconnect from that client. These two methods only
59/// work once a named pipe is associated with an event loop.
60///
61/// The `connect` method will succeed asynchronously and a completion can be
62/// detected once the object receives a writable notification.
63///
64/// # Named pipe clients
65///
66/// Currently to create a client of a named pipe server then you can use the
67/// `OpenOptions` type in the standard library to create a `File` that connects
68/// to a named pipe. Afterwards you can use the `into_raw_handle` method coupled
69/// with the `NamedPipe::from_raw_handle` method to convert that to a named pipe
70/// that can operate asynchronously. Don't forget to pass the
71/// `FILE_FLAG_OVERLAPPED` flag when opening the `File`.
72pub struct NamedPipe {
73    inner: Arc<Inner>,
74}

132    /// Issue a connection request with the specified overlapped operation.
133    ///
134    /// This function will issue a request to connect a client to this server,
135    /// returning immediately after starting the overlapped operation.
136    ///
137    /// If this function immediately succeeds then `Ok(true)` is returned. If
138    /// the overlapped operation is enqueued and pending, then `Ok(false)` is
139    /// returned. Otherwise an error is returned indicating what went wrong.
140    ///
141    /// # Unsafety
142    ///
143    /// This function is unsafe because the kernel requires that the
144    /// `overlapped` pointer is valid until the end of the I/O operation. The
145    /// kernel also requires that `overlapped` is unique for this I/O operation
146    /// and is not in use for any other I/O.
147    ///
148    /// To safely use this function callers must ensure that this pointer is
149    /// valid until the I/O operation is completed, typically via completion
150    /// ports and waiting to receive the completion notification on the port.
151    pub unsafe fn connect_overlapped(&self, overlapped: *mut OVERLAPPED) -> io::Result<bool> {
152        if ConnectNamedPipe(self.handle.raw(), overlapped) != 0 {
153            return Ok(true);
154        }
155
156        let err = io::Error::last_os_error();
157
158        match err.raw_os_error().map(|e| e as u32) {
159            Some(ERROR_PIPE_CONNECTED) => Ok(true),
160            Some(ERROR_NO_DATA) => Ok(true),
161            Some(ERROR_IO_PENDING) => Ok(false),
162            _ => Err(err),
163        }
164    }
165
166    /// Disconnects this named pipe from any connected client.
167    pub fn disconnect(&self) -> io::Result<()> {
168        if unsafe { DisconnectNamedPipe(self.handle.raw()) } == 0 {
169            Err(io::Error::last_os_error())
170        } else {
171            Ok(())
172        }
173    }

175    /// Issues an overlapped read operation to occur on this pipe.
176    ///
177    /// This function will issue an asynchronous read to occur in an overlapped
178    /// fashion, returning immediately. The `buf` provided will be filled in
179    /// with data and the request is tracked by the `overlapped` function
180    /// provided.
181    ///
182    /// If the operation succeeds immediately, `Ok(Some(n))` is returned where
183    /// `n` is the number of bytes read. If an asynchronous operation is
184    /// enqueued, then `Ok(None)` is returned. Otherwise if an error occurred
185    /// it is returned.
186    ///
187    /// When this operation completes (or if it completes immediately), another
188    /// mechanism must be used to learn how many bytes were transferred (such as
189    /// looking at the filed in the IOCP status message).
190    ///
191    /// # Unsafety
192    ///
193    /// This function is unsafe because the kernel requires that the `buf` and
194    /// `overlapped` pointers to be valid until the end of the I/O operation.
195    /// The kernel also requires that `overlapped` is unique for this I/O
196    /// operation and is not in use for any other I/O.
197    ///
198    /// To safely use this function callers must ensure that the pointers are
199    /// valid until the I/O operation is completed, typically via completion
200    /// ports and waiting to receive the completion notification on the port.
201    pub unsafe fn read_overlapped(
202        &self,
203        buf: &mut [u8],
204        overlapped: *mut OVERLAPPED,
205    ) -> io::Result<Option<usize>> {
206        let len = std::cmp::min(buf.len(), u32::MAX as usize) as u32;
207        let res = ReadFile(
208            self.handle.raw(),
209            buf.as_mut_ptr() as *mut _,
210            len,
211            std::ptr::null_mut(),
212            overlapped,
213        );
214        if res == 0 {
215            let err = io::Error::last_os_error();
216            if err.raw_os_error() != Some(ERROR_IO_PENDING as i32) {
217                return Err(err);
218            }
219        }
220
221        let mut bytes = 0;
222        let res = GetOverlappedResult(self.handle.raw(), overlapped, &mut bytes, 0);
223        if res == 0 {
224            let err = io::Error::last_os_error();
225            if err.raw_os_error() == Some(ERROR_IO_INCOMPLETE as i32) {
226                Ok(None)
227            } else {
228                Err(err)
229            }
230        } else {
231            Ok(Some(bytes as usize))
232        }
233    }

263        buf: &[u8],
264        overlapped: *mut OVERLAPPED,

267        let res = WriteFile(
268            self.handle.raw(),
269            buf.as_ptr() as *const _,
270            len,
271            std::ptr::null_mut(),
272            overlapped,
273        );
274        if res == 0 {
275            let err = io::Error::last_os_error();
276            if err.raw_os_error() != Some(ERROR_IO_PENDING as i32) {
277                return Err(err);
278            }
279        }
280
281        let mut bytes = 0;
282        let res = GetOverlappedResult(self.handle.raw(), overlapped, &mut bytes, 0);
283        if res == 0 {
284            let err = io::Error::last_os_error();
285            if err.raw_os_error() == Some(ERROR_IO_INCOMPLETE as i32) {
286                Ok(None)
287            } else {
288                Err(err)
289            }
290        } else {
291            Ok(Some(bytes as usize))
292        }
293    }

295    /// Calls the `GetOverlappedResult` function to get the result of an
296    /// overlapped operation for this handle.
297    ///
298    /// This function takes the `OVERLAPPED` argument which must have been used
299    /// to initiate an overlapped I/O operation, and returns either the
300    /// successful number of bytes transferred during the operation or an error
301    /// if one occurred.
302    ///
303    /// # Unsafety
304    ///
305    /// This function is unsafe as `overlapped` must have previously been used
306    /// to execute an operation for this handle, and it must also be a valid
307    /// pointer to an `Overlapped` instance.
308    #[inline]
309    unsafe fn result(&self, overlapped: *mut OVERLAPPED) -> io::Result<usize> {
310        let mut transferred = 0;
311        let r = GetOverlappedResult(self.handle.raw(), overlapped, &mut transferred, 0);
312        if r == 0 {
313            Err(io::Error::last_os_error())
314        } else {
315            Ok(transferred as usize)
316        }
317    }

369impl NamedPipe {
370    /// Creates a new named pipe at the specified `addr` given a "reasonable
371    /// set" of initial configuration options.
372    pub fn new<A: AsRef<OsStr>>(addr: A) -> io::Result<NamedPipe> {
373        use std::os::windows::ffi::OsStrExt;
374        let name: Vec<_> = addr.as_ref().encode_wide().chain(Some(0)).collect();
375
376        // Safety: syscall
377        let h = unsafe {
378            CreateNamedPipeW(
379                name.as_ptr(),
380                PIPE_ACCESS_DUPLEX | FILE_FLAG_FIRST_PIPE_INSTANCE | FILE_FLAG_OVERLAPPED,
381                PIPE_TYPE_BYTE,
382                PIPE_UNLIMITED_INSTANCES,
383                65536,
384                65536,
385                0,
386                std::ptr::null_mut(),
387            )
388        };
389
390        if h == INVALID_HANDLE_VALUE {
391            Err(io::Error::last_os_error())
392        } else {
393            // Safety: nothing actually unsafe about this. The trait fn includes
394            // `unsafe`.
395            Ok(unsafe { Self::from_raw_handle(h as RawHandle) })
396        }
397    }
398
399    /// Attempts to call `ConnectNamedPipe`, if possible.
400    ///
401    /// This function will attempt to connect this pipe to a client in an
402    /// asynchronous fashion. If the function immediately establishes a
403    /// connection to a client then `Ok(())` is returned. Otherwise if a
404    /// connection attempt was issued and is now in progress then a "would
405    /// block" error is returned.
406    ///
407    /// When the connection is finished then this object will be flagged as
408    /// being ready for a write, or otherwise in the writable state.
409    ///
410    /// # Errors
411    ///
412    /// This function will return a "would block" error if the pipe has not yet
413    /// been registered with an event loop, if the connection operation has
414    /// previously been issued but has not yet completed, or if the connect
415    /// itself was issued and didn't finish immediately.
416    ///
417    /// Normal I/O errors from the call to `ConnectNamedPipe` are returned
418    /// immediately.
419    pub fn connect(&self) -> io::Result<()> {
420        // "Acquire the connecting lock" or otherwise just make sure we're the
421        // only operation that's using the `connect` overlapped instance.
422        if self.inner.connecting.swap(true, SeqCst) {
423            return Err(would_block());
424        }
425
426        // Now that we've flagged ourselves in the connecting state, issue the
427        // connection attempt. Afterwards interpret the return value and set
428        // internal state accordingly.
429        let res = unsafe {
430            let overlapped = self.inner.connect.as_ptr() as *mut _;
431            self.inner.connect_overlapped(overlapped)
432        };
433
434        match res {
435            // The connection operation finished immediately, so let's schedule
436            // reads/writes and such.
437            Ok(true) => {
438                self.inner.connecting.store(false, SeqCst);
439                Inner::post_register(&self.inner, None);
440                Ok(())
441            }
442
443            // If the overlapped operation was successful and didn't finish
444            // immediately then we forget a copy of the arc we hold
445            // internally. This ensures that when the completion status comes
446            // in for the I/O operation finishing it'll have a reference
447            // associated with it and our data will still be valid. The
448            // `connect_done` function will "reify" this forgotten pointer to
449            // drop the refcount on the other side.
450            Ok(false) => {
451                mem::forget(self.inner.clone());
452                Err(would_block())
453            }
454
455            Err(e) => {
456                self.inner.connecting.store(false, SeqCst);
457                Err(e)
458            }
459        }
460    }
461
462    /// Takes any internal error that has happened after the last I/O operation
463    /// which hasn't been retrieved yet.
464    ///
465    /// This is particularly useful when detecting failed attempts to `connect`.
466    /// After a completed `connect` flags this pipe as writable then callers
467    /// must invoke this method to determine whether the connection actually
468    /// succeeded. If this function returns `None` then a client is connected,
469    /// otherwise it returns an error of what happened and a client shouldn't be
470    /// connected.
471    pub fn take_error(&self) -> io::Result<Option<io::Error>> {
472        Ok(self.inner.io.lock().unwrap().connect_error.take())
473    }
474
475    /// Disconnects this named pipe from a connected client.
476    ///
477    /// This function will disconnect the pipe from a connected client, if any,
478    /// transitively calling the `DisconnectNamedPipe` function.
479    ///
480    /// After a `disconnect` is issued, then a `connect` may be called again to
481    /// connect to another client.
482    pub fn disconnect(&self) -> io::Result<()> {
483        self.inner.disconnect()
484    }
485}

526impl<'a> Read for &'a NamedPipe {
527    fn read(&mut self, buf: &mut [u8]) -> io::Result<usize> {
528        let mut state = self.inner.io.lock().unwrap();
529
530        if state.token.is_none() {
531            return Err(would_block());
532        }
533
534        match mem::replace(&mut state.read, State::None) {
535            // In theory not possible with `token` checked above,
536            // but return would block for now.
537            State::None => Err(would_block()),
538
539            // A read is in flight, still waiting for it to finish
540            State::Pending(buf, amt) => {
541                state.read = State::Pending(buf, amt);
542                Err(would_block())
543            }
544
545            // We previously read something into `data`, try to copy out some
546            // data. If we copy out all the data schedule a new read and
547            // otherwise store the buffer to get read later.
548            State::Ok(data, cur) => {
549                let n = {
550                    let mut remaining = &data[cur..];
551                    remaining.read(buf)?
552                };
553                let next = cur + n;
554                if next != data.len() {
555                    state.read = State::Ok(data, next);
556                } else {
557                    self.inner.put_buffer(data);
558                    Inner::schedule_read(&self.inner, &mut state, None);
559                }
560                Ok(n)
561            }
562
563            // Looks like an in-flight read hit an error, return that here while
564            // we schedule a new one.
565            State::Err(e) => {
566                Inner::schedule_read(&self.inner, &mut state, None);
567                if e.raw_os_error() == Some(ERROR_BROKEN_PIPE as i32) {
568                    Ok(0)
569                } else {
570                    Err(e)
571                }
572            }
573        }
574    }
575}

577impl<'a> Write for &'a NamedPipe {
578    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
579        // Make sure there's no writes pending
580        let mut io = self.inner.io.lock().unwrap();
581
582        if io.token.is_none() {
583            return Err(would_block());
584        }
585
586        match io.write {
587            State::None => {}
588            State::Err(_) => match mem::replace(&mut io.write, State::None) {
589                State::Err(e) => return Err(e),
590                // `io` is locked, so this branch is unreachable
591                _ => unreachable!(),
592            },
593            // any other state should be handled in `write_done`
594            _ => {
595                return Err(would_block());
596            }
597        }
598
599        // Move `buf` onto the heap and fire off the write
600        let mut owned_buf = self.inner.get_buffer();
601        owned_buf.extend(buf);
602        match Inner::maybe_schedule_write(&self.inner, owned_buf, 0, &mut io)? {
603            // Some bytes are written immediately
604            Some(n) => Ok(n),
605            // Write operation is enqueued for whole buffer
606            None => Ok(buf.len()),
607        }
608    }
609
610    fn flush(&mut self) -> io::Result<()> {
611        Ok(())
612    }
613}

615impl Source for NamedPipe {
616    fn register(&mut self, registry: &Registry, token: Token, _: Interest) -> io::Result<()> {
617        let mut io = self.inner.io.lock().unwrap();
618
619        io.check_association(registry, false)?;
620
621        if io.token.is_some() {
622            return Err(io::Error::new(
623                io::ErrorKind::AlreadyExists,
624                "I/O source already registered with a `Registry`",
625            ));
626        }
627
628        if io.cp.is_none() {
629            let selector = registry.selector();
630
631            io.cp = Some(selector.clone_port());
632
633            let inner_token = NEXT_TOKEN.fetch_add(2, Relaxed) + 2;
634            selector.inner.cp.add_handle(inner_token, self)?;
635        }
636
637        io.token = Some(token);
638        drop(io);
639
640        Inner::post_register(&self.inner, None);
641
642        Ok(())
643    }
644
645    fn reregister(&mut self, registry: &Registry, token: Token, _: Interest) -> io::Result<()> {
646        let mut io = self.inner.io.lock().unwrap();
647
648        io.check_association(registry, true)?;
649
650        io.token = Some(token);
651        drop(io);
652
653        Inner::post_register(&self.inner, None);
654
655        Ok(())
656    }
657
658    fn deregister(&mut self, registry: &Registry) -> io::Result<()> {
659        let mut io = self.inner.io.lock().unwrap();
660
661        io.check_association(registry, true)?;
662
663        if io.token.is_none() {
664            return Err(io::Error::new(
665                io::ErrorKind::NotFound,
666                "I/O source not registered with `Registry`",
667            ));
668        }
669
670        io.token = None;
671        Ok(())
672    }
673}

687impl Drop for NamedPipe {
688    fn drop(&mut self) {
689        // Cancel pending reads/connects, but don't cancel writes to ensure that
690        // everything is flushed out.
691        unsafe {
692            if self.inner.connecting.load(SeqCst) {
693                drop(cancel(&self.inner.handle, &self.inner.connect));
694            }
695
696            let io = self.inner.io.lock().unwrap();
697            if let State::Pending(..) = io.read {
698                drop(cancel(&self.inner.handle, &self.inner.read));
699            }
700        }
701    }
702}

704impl Inner {
705    /// Schedules a read to happen in the background, executing an overlapped
706    /// operation.
707    ///
708    /// This function returns `true` if either of the following conditions are met:
709    /// * A normal error happens
710    /// * The read is scheduled in the background
711    /// * Data is already available to be read (ERROR_MORE_DATA)
712    ///
713    /// If the pipe is no longer connected (ERROR_PIPE_LISTENING) then `false` is
714    /// returned and no read is scheduled.
715    fn schedule_read(me: &Arc<Inner>, io: &mut Io, events: Option<&mut Vec<Event>>) -> bool {
716        // Check to see if a read is already scheduled/completed
717        match io.read {
718            State::None => {}
719            _ => return true,
720        }
721
722        // Allocate a buffer and schedule the read.
723        let mut buf = me.get_buffer();
724        let e = unsafe {
725            let overlapped = me.read.as_ptr() as *mut _;
726            let slice = slice::from_raw_parts_mut(buf.as_mut_ptr(), buf.capacity());
727            me.read_overlapped(slice, overlapped)
728        };
729
730        match e {
731            // See `NamedPipe::connect` above for the rationale behind `forget`
732            Ok(_) => {
733                io.read = State::Pending(buf, 0); // 0 is ignored on read side
734                mem::forget(me.clone());
735                true
736            }
737
738            // If ERROR_PIPE_LISTENING happens then it's not a real read error,
739            // we just need to wait for a connect.
740            Err(ref e) if e.raw_os_error() == Some(ERROR_PIPE_LISTENING as i32) => false,
741
742            // If ERROR_MORE_DATA is returned, it means the slice of unused capacity of the
743            // buffer provided is less than the amount of data available to be read. So
744            // prioritize draining the buffer before scheduling a new read.
745            //
746            // Return `true` to indicate that an overlapped read was scheduled "successfully",
747            // without actually scheduling it. Instead, update `io.read` to `State::Ok(buf, 0)`
748            // to ensure that the next `std::io::Read::read` call is presented still with the
749            // unread data to read from.
750            Err(ref e) if e.raw_os_error() == Some(ERROR_MORE_DATA as i32) => {
751                io.read = State::Ok(buf, 0);
752                mem::forget(me.clone());
753                true
754            }
755
756            // If some other error happened, though, we're now readable to give
757            // out the error.
758            Err(e) => {
759                io.read = State::Err(e);
760                io.notify_readable(me, events);
761                true
762            }
763        }
764    }
765
766    /// Maybe schedules overlapped write operation.
767    ///
768    /// * `None` means that overlapped operation was enqueued
769    /// * `Some(n)` means that `n` bytes was immediately written.
770    ///   Note, that `write_done` will fire anyway to clean up the state.
771    fn maybe_schedule_write(
772        me: &Arc<Inner>,
773        buf: Vec<u8>,
774        pos: usize,
775        io: &mut Io,
776    ) -> io::Result<Option<usize>> {
777        // Very similar to `schedule_read` above, just done for the write half.
778        let e = unsafe {
779            let overlapped = me.write.as_ptr() as *mut _;
780            me.write_overlapped(&buf[pos..], overlapped)
781        };
782
783        // See `connect` above for the rationale behind `forget`
784        match e {
785            // `n` bytes are written immediately
786            Ok(Some(n)) => {
787                io.write = State::Ok(buf, pos);
788                mem::forget(me.clone());
789                Ok(Some(n))
790            }
791            // write operation is enqueued
792            Ok(None) => {
793                io.write = State::Pending(buf, pos);
794                mem::forget(me.clone());
795                Ok(None)
796            }
797            Err(e) => Err(e),
798        }
799    }
800
801    fn schedule_write(
802        me: &Arc<Inner>,
803        buf: Vec<u8>,
804        pos: usize,
805        io: &mut Io,
806        events: Option<&mut Vec<Event>>,
807    ) {
808        match Inner::maybe_schedule_write(me, buf, pos, io) {
809            Ok(Some(_)) => {
810                // immediate result will be handled in `write_done`,
811                // so we'll reinterpret the `Ok` state
812                let state = mem::replace(&mut io.write, State::None);
813                io.write = match state {
814                    State::Ok(buf, pos) => State::Pending(buf, pos),
815                    // io is locked, so this branch is unreachable
816                    _ => unreachable!(),
817                };
818                mem::forget(me.clone());
819            }
820            Ok(None) => (),
821            Err(e) => {
822                io.write = State::Err(e);
823                io.notify_writable(me, events);
824            }
825        }
826    }
827
828    fn post_register(me: &Arc<Inner>, mut events: Option<&mut Vec<Event>>) {
829        let mut io = me.io.lock().unwrap();
830        #[allow(clippy::needless_option_as_deref)]
831        if Inner::schedule_read(me, &mut io, events.as_deref_mut()) {
832            if let State::None = io.write {
833                io.notify_writable(me, events);
834            }
835        }
836    }
837
838    fn get_buffer(&self) -> Vec<u8> {
839        self.pool.lock().unwrap().get(4 * 1024)
840    }
841
842    fn put_buffer(&self, buf: Vec<u8>) {
843        self.pool.lock().unwrap().put(buf)
844    }
845}

847unsafe fn cancel(handle: &Handle, overlapped: &Overlapped) -> io::Result<()> {
848    let ret = CancelIoEx(handle.raw(), overlapped.as_ptr());
849    // `CancelIoEx` returns 0 on error:
850    // https://docs.microsoft.com/en-us/windows/win32/fileio/cancelioex-func
851    if ret == 0 {
852        Err(io::Error::last_os_error())
853    } else {
854        Ok(())
855    }
856}

858fn connect_done(status: &OVERLAPPED_ENTRY, events: Option<&mut Vec<Event>>) {
859    let status = CompletionStatus::from_entry(status);
860
861    // Acquire the `Arc<Inner>`. Note that we should be guaranteed that
862    // the refcount is available to us due to the `mem::forget` in
863    // `connect` above.
864    let me = unsafe { Arc::from_raw(Inner::ptr_from_conn_overlapped(status.overlapped())) };
865
866    // Flag ourselves as no longer using the `connect` overlapped instances.
867    let prev = me.connecting.swap(false, SeqCst);
868    assert!(prev, "NamedPipe was not previously connecting");
869
870    // Stash away our connect error if one happened
871    debug_assert_eq!(status.bytes_transferred(), 0);
872    unsafe {
873        match me.result(status.overlapped()) {
874            Ok(n) => debug_assert_eq!(n, 0),
875            Err(e) => me.io.lock().unwrap().connect_error = Some(e),
876        }
877    }
878
879    // We essentially just finished a registration, so kick off a
880    // read and register write readiness.
881    Inner::post_register(&me, events);
882}
883
884fn read_done(status: &OVERLAPPED_ENTRY, events: Option<&mut Vec<Event>>) {
885    let status = CompletionStatus::from_entry(status);
886
887    // Acquire the `FromRawArc<Inner>`. Note that we should be guaranteed that
888    // the refcount is available to us due to the `mem::forget` in
889    // `schedule_read` above.
890    let me = unsafe { Arc::from_raw(Inner::ptr_from_read_overlapped(status.overlapped())) };
891
892    let mut io = me.io.lock().unwrap();
893    let mut buf = match mem::replace(&mut io.read, State::None) {
894        State::Ok(buf, pos) => {
895            io.read = State::Ok(buf, pos);
896
897            // Flag readiness that we have undelivered data to be read.
898            io.notify_readable(&me, events);
899            return;
900        }
901        State::Pending(buf, _) => buf,
902        State::Err(e) => {
903            io.read = State::Err(e);
904
905            // Flag readiness that the error needs to be delivered.
906            io.notify_readable(&me, events);
907            return;
908        }
909        State::None => unreachable!(),
910    };
911    unsafe {
912        match me.result(status.overlapped()) {
913            Ok(n) => {
914                debug_assert_eq!(status.bytes_transferred() as usize, n);
915                buf.set_len(status.bytes_transferred() as usize);
916                io.read = State::Ok(buf, 0);
917            }
918            // This is non-fatal. The buffer was simply too small for the entire message.
919            // Deliver the bytes we got, and if the caller wants to read the rest of the
920            // message, they can initiate another read.
921            Err(e) if e.raw_os_error() == Some(ERROR_MORE_DATA as i32) => {
922                buf.set_len(status.bytes_transferred() as usize);
923                io.read = State::Ok(buf, 0);
924            }
925            Err(e) => {
926                debug_assert_eq!(status.bytes_transferred(), 0);
927                io.read = State::Err(e);
928            }
929        }
930    }
931
932    // Flag our readiness that we've got data.
933    io.notify_readable(&me, events);
934}
935
936fn write_done(status: &OVERLAPPED_ENTRY, events: Option<&mut Vec<Event>>) {
937    let status = CompletionStatus::from_entry(status);
938
939    // Acquire the `Arc<Inner>`. Note that we should be guaranteed that
940    // the refcount is available to us due to the `mem::forget` in
941    // `schedule_write` above.
942    let me = unsafe { Arc::from_raw(Inner::ptr_from_write_overlapped(status.overlapped())) };
943
944    // Make the state change out of `Pending`. If we wrote the entire buffer
945    // then we're writable again and otherwise we schedule another write.
946    let mut io = me.io.lock().unwrap();
947    let (buf, pos) = match mem::replace(&mut io.write, State::None) {
948        // `Ok` here means, that the operation was completed immediately
949        // `bytes_transferred` is already reported to a client.
950        // Hence, we don't reset the state to `Ok` but leave it in `None`.
951        State::Ok(..) => {
952            io.notify_writable(&me, events);
953            return;
954        }
955        State::Pending(buf, pos) => (buf, pos),
956        State::Err(e) => {
957            io.write = State::Err(e);
958
959            // Flag readiness that the error needs to be delivered.
960            io.notify_writable(&me, events);
961            return;
962        }
963        State::None => unreachable!(),
964    };
965
966    unsafe {
967        match me.result(status.overlapped()) {
968            Ok(n) => {
969                debug_assert_eq!(status.bytes_transferred() as usize, n);
970                let new_pos = pos + (status.bytes_transferred() as usize);
971                if new_pos == buf.len() {
972                    me.put_buffer(buf);
973                    io.notify_writable(&me, events);
974                } else {
975                    Inner::schedule_write(&me, buf, new_pos, &mut io, events);
976                }
977            }
978            Err(e) => {
979                debug_assert_eq!(status.bytes_transferred(), 0);
980                io.write = State::Err(e);
981                io.notify_writable(&me, events);
982            }
983        }
984    }
985}

987fn event_done(status: &OVERLAPPED_ENTRY, events: Option<&mut Vec<Event>>) {
988    let status = CompletionStatus::from_entry(status);
989
990    // Acquire the `Arc<Inner>`. Note that we should be guaranteed that
991    // the refcount is available to us due to the `mem::forget` in
992    // `schedule_write` above.
993    let me = unsafe { Arc::from_raw(Inner::ptr_from_event_overlapped(status.overlapped())) };
994
995    let io = me.io.lock().unwrap();
996
997    // Make sure the I/O handle is still registered with the selector
998    if io.token.is_some() {
999        // This method is also called during `Selector::drop` to perform
1000        // cleanup. In this case, `events` is `None` and we don't need to track
1001        // the event.
1002        if let Some(events) = events {
1003            let mut ev = Event::from_completion_status(&status);
1004            // Reverse the `.data` alteration done in `schedule_event`. This
1005            // alteration was done so the selector recognized the event as one from
1006            // a named pipe.
1007            ev.data >>= 1;
1008            events.push(ev);
1009        }
1010    }
1011}

1013impl Io {
1014    fn check_association(&self, registry: &Registry, required: bool) -> io::Result<()> {
1015        match self.cp {
1016            Some(ref cp) if !registry.selector().same_port(cp) => Err(io::Error::new(
1017                io::ErrorKind::AlreadyExists,
1018                "I/O source already registered with a different `Registry`",
1019            )),
1020            None if required => Err(io::Error::new(
1021                io::ErrorKind::NotFound,
1022                "I/O source not registered with `Registry`",
1023            )),
1024            _ => Ok(()),
1025        }
1026    }
1027
1028    fn notify_readable(&self, me: &Arc<Inner>, events: Option<&mut Vec<Event>>) {
1029        if let Some(token) = self.token {
1030            let mut ev = Event::new(token);
1031            ev.set_readable();
1032
1033            if let Some(events) = events {
1034                events.push(ev);
1035            } else {
1036                self.schedule_event(me, ev);
1037            }
1038        }
1039    }
1040
1041    fn notify_writable(&self, me: &Arc<Inner>, events: Option<&mut Vec<Event>>) {
1042        if let Some(token) = self.token {
1043            let mut ev = Event::new(token);
1044            ev.set_writable();
1045
1046            if let Some(events) = events {
1047                events.push(ev);
1048            } else {
1049                self.schedule_event(me, ev);
1050            }
1051        }
1052    }
1053
1054    fn schedule_event(&self, me: &Arc<Inner>, mut event: Event) {
1055        // Alter the token so that the selector will identify the IOCP event as
1056        // one for a named pipe. This will be reversed in `event_done`
1057        //
1058        // `data` for named pipes is an auto-incrementing counter. Because
1059        // `data` is `u64` we do not risk losing the most-significant bit
1060        // (unless a user creates 2^62 named pipes during the lifetime of the
1061        // process).
1062        event.data <<= 1;
1063        event.data += 1;
1064
1065        let completion_status =
1066            event.to_completion_status_with_overlapped(me.event.as_ptr() as *mut _);
1067
1068        match self.cp.as_ref().unwrap().post(completion_status) {
1069            Ok(_) => {
1070                // Increase the ref count of `Inner` for the completion event.
1071                mem::forget(me.clone());
1072            }
1073            Err(_) => {
1074                // Nothing to do here
1075            }
1076        }
1077    }
1078}

1082}
1083
1084impl BufferPool {
1085    fn with_capacity(cap: usize) -> BufferPool {
1086        BufferPool {
1087            pool: Vec::with_capacity(cap),
1088        }
1089    }
1090
1091    fn get(&mut self, default_cap: usize) -> Vec<u8> {
1092        self.pool
1093            .pop()
1094            .unwrap_or_else(|| Vec::with_capacity(default_cap))
1095    }
1096
1097    fn put(&mut self, mut buf: Vec<u8>) {
1098        if self.pool.len() < self.pool.capacity() {
1099            unsafe {
1100                buf.set_len(0);
1101            }
1102            self.pool.push(buf);
1103        }
1104    }
