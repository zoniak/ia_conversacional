1use reqwest::Client;
2use serde_json::{Value, json}; // Importamos Value para poder tipar nuestro vector
3use std::env;
4use std::fs;
5use std::io::{self, Write}; // <-- NUEVO: Para interactuar con archivos
6
7// Usamos Tokio para permitir funciones asíncronas
8#[tokio::main]
9async fn main() -> Result<(), Box<dyn std::error::Error>> {
10    // 1. Cargar la variable de entorno de forma segura
11    dotenvy::dotenv().ok();
12    // Modifica la línea donde se lee la clave añadiendo .trim()
13    let api_key = env::var("GEMINI_API_KEY")
14        .expect("¡Falta la variable GEMINI_API_KEY en el archivo .env!")
15        .trim() // <-- NUEVO: Elimina espacios, tabulaciones y saltos de línea ocultos
16        .to_string();
17
18    // 2. Configurar el cliente HTTP y la URL (usamos el modelo flash por ser rápido y económico)
19    let client = Client::new();
20    // AHORA (Modelo actualizado y activo):
21    let url = format!(
22        "https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key={}",
23        api_key
24    );
25
26    // Imprimimos los primeros 10 caracteres de la clave y ocultamos el resto por seguridad
27    let safe_url = url.replace(&api_key[10..], "********");
28    println!(
29        "🔍 [DEBUG] La URL exacta que estamos llamando es:\n{}\n",
30        safe_url
31    );
32
33    println!("========================================");
34    println!("Bienvenido al Chat con Gemini en Rust");
35    println!("(Escribe 'salir' para terminar el programa)");
36    println!("========================================\n");
37
38    // 3. Bucle infinito para mantener el chat abierto
39    // NUEVO: Creamos un Vector mutable para almacenar el historial de la conversación.
40    let mut history: Vec<Value> = Vec::new();
41
42    loop {
43        print!("Tú: ");
44        io::stdout().flush()?;
45
46        let mut input = String::new();
47        io::stdin().read_line(&mut input)?;
48        let input = input.trim();
49
50        if input.eq_ignore_ascii_case("salir") {
51            println!("¡Hasta luego!");
52            break;
53        }
54
55        if input.is_empty() {
56            continue;
57        }
58
59        // 1. Añadimos el mensaje actual del usuario al historial
60        history.push(json!({
61            "role": "user",
62            "parts": [{"text": input}]
63        }));
64
65        // 2. Construimos el cuerpo de la petición enviando TODO el historial
66        let body = json!({
67            "contents": history
68        });
69
70        // 3. Enviamos la petición
71        let response = client.post(&url).json(&body).send().await?;
72
73        if response.status().is_success() {
74            let json_res: Value = response.json().await?;
75
76            // Extraemos la respuesta
77            if let Some(text) = json_res["candidates"][0]["content"]["parts"][0]["text"].as_str() {
78                let response_text = text.trim();
79                println!("Gemini: {}\n", response_text);
80
81                // 4. NUEVO: Añadimos la respuesta de Gemini al historial
82                history.push(json!({
83                    "role": "model",
84                    "parts": [{"text": response_text}]
85                }));
86
87                // ==========================================
88                // NUEVO: GUARDAR EL HISTORIAL EN UN ARCHIVO
89                // ==========================================
90                // 1. Convertimos el vector 'history' a un texto JSON formateado (bonito y legible)
91                let historial_json = serde_json::to_string_pretty(&history)?;
92
93                // 2. Lo escribimos en un archivo. Si no existe, lo crea. Si existe, lo sobrescribe.
94                fs::write("historial_chat.json", historial_json)?;
95                // ==========================================
96
97                //Cuando no quiera guardar el historial eliminar las lineas 1. y 2.
98            } else {
99                println!("Gemini: [No se pudo extraer el texto de la respuesta]\n");
100                // Si algo falla al procesar, sacamos el último mensaje del usuario
101                // para no romper la alternancia "user" -> "model"
102                history.pop();
103            }
104        } else {
105            println!("Error en la petición: {:?}\n", response.status());
106            // Si la API da error, retiramos el último mensaje del usuario del historial
107            history.pop();
108        }
109    }
110
111    Ok(())
112}
